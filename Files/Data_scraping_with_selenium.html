Yet another option we have for getting financial data from the internet is scrapping a website such as financialmodelingprep.com This website contains several tables within the same address but those tables aren't visible until you click on the on the screen. This is because the html contains javascripts that produce the table on the fly.

For our purposes this means that we need to use a webdriver to automate the click on a table. I will be using chromedriver.

The function defined below uses chromedriver to open either the yealy or quarterly financial data on financialmodelingprep.com and goes through three sets of tables 'Income Statement', 'Balance Sheet Statement', 'Cash Flow Statement' retrieving the tables with pandas read_html.

def get_financials(ticker=None, quarterly=False,
                   chrome_path=r"C:\Users\victor.sequi\Downloads\chromedriver_win32\chromedriver.exe"):
    driver = webdriver.Chrome(chrome_path)
    if quarterly:
        url = f'https://financialmodelingprep.com/financial-statements-quarterly/{ticker}'
    else:
        url = f'https://financialmodelingprep.com/financial-statements/{ticker}'

    page = driver.get(url)
    Financials = []
    Financials_names = ['Income Statement', 'Retained Earning Schedule', 'PPE Schedule',
                        'Intangible and Goodwill Schedule', 'Balance Sheet Statement',
                        'Retained Earning Schedule', 'PPE Schedule', 'Intangible and Goodwill Schedule',
                        'Cash Flow Statement', 'Retained Earning Schedule', 'PPE Schedule',
                        'Intangible and Goodwill Schedule']

    for Hyper_link in driver.find_elements_by_class_name('item'):
        Hyper_link.click()
        for i, table in enumerate(pd.read_html(driver.page_source)):
            Financials.append(table)
        time.sleep(2)
    driver.close()
    Financials = dict(zip(Financials_names, Financials))
    response = requests.get(url)
    results_page = BeautifulSoup(response.content, 'lxml')
    Financials['price'] = float(results_page.find('div',
                                                  attrs={'class': 'company-price'}).contents[0].strip().strip('$'))
    return Financials

Rev_stocks = pd.read_csv("../01 Data/RevStock.csv", encoding="ISO-8859-1")
All_Yearly_financials = {}


for ticker in tqdm(Rev_stocks.Ticker[:10]):
    Yearly_Financials = get_financials(ticker)
    All_Yearly_financials[ticker] = Yearly_Financials
    now = datetime.now().strftime("%d-%m-%y")
    file_to_write = bz2.BZ2File(f"../01 Data/stocks_dicts/{ticker}_{now}_Yearly_financials.pkl", "wb")
    pickle.dump(Yearly_Financials, file_to_write)

def key_financials(ticker_dict):
    key_fin = pd.DataFrame()
    key_fin['Shares Outstanding Diluted'] = transpose(ticker_dict)['Weighted Average Shares Outstanding (Diluted)']
    key_fin['EPS Diluted'] = transpose(ticker_dict)['Net Income'] / key_fin['Shares Outstanding Diluted']
    key_fin['P/E Diluted'] = ticker_dict['price'] / key_fin['EPS Diluted']
    key_fin['Dividend per share'] = -transpose(ticker_dict, table='Cash Flow Statement')['Dividend payments'] / key_fin['Shares Outstanding Diluted']
    key_fin['Debt/Equity'] = transpose(ticker_dict, table='Balance Sheet Statement')['Total debt'] / transpose(ticker_dict, table='Balance Sheet Statement')['Total shareholders equity']
    key_fin['BV Diluted'] = transpose(ticker_dict, table='Balance Sheet Statement')['Total shareholders equity'] / key_fin['Shares Outstanding Diluted']
    key_fin['P/BV Diluted'] = ticker_dict['price'] / key_fin['BV Diluted']
    key_fin['ROE'] = transpose(ticker_dict)['Net Income'] / transpose(ticker_dict, table='Balance Sheet Statement')['Total shareholders equity']
    key_fin['Current Ratio'] = transpose(ticker_dict, table='Balance Sheet Statement')['Total assets'] / transpose(ticker_dict, table='Balance Sheet Statement')['Total liabilities']
    key_fin['Retained Earnings Diff'] = transpose(All_Yearly_financials['AAPL'], table='Retained Earning Schedule')['Retained Earnings'] - transpose(All_Yearly_financials['AAPL'], table='Retained Earning Schedule')['Retained Earnings (Previous Year)']
    key_fin.sort_index(inplace=True)
    return key_fin

def plots(key_fin, cols_to_plot=['EPS Diluted', 'BV Diluted']):

    ax = key_fin[cols_to_plot].plot(figsize=(15, 10),
                                    title=' - '.join(cols_to_plot), marker='o')

    for col in cols_to_plot:
        for i, y in enumerate(key_fin[col]):
            x = i
            y = key_fin[col][i]
            label = "{:.2f}".format(y)
            ax.annotate(label,  # this is the text
                        (x, y),  # this is the point to label
                        textcoords="offset pixels",  # how to position the text
                        xytext=(0, 10),  # distance from text to points (x,y)
                        ha='center')  # horizontal alignment can be left, right or center
            ax.set_xlabel("Year", fontsize=15)
    display(ax)
